SELECT * FROM `tarasov_univer`.`begunok`;

SELECT student_idstudent, mark FROM `tarasov_univer`.`begunok`;

SELECT 
	`student`.`surname`, `student`.`name`, mark, `prepod`.`surname`, `prepod`.`name`
FROM 
	`tarasov_univer`.`student`,
    `tarasov_univer`.`begunok`,
    `tarasov_univer`.`prepod`
WHERE
	`student`.`idstudent` = `begunok`.`student_idstudent`
    AND `prepod`.`idprepod` = `begunok`.`prepod_idprepod`
    AND `begunok`.`mark` in ('3')
